kotlin-stdlib / kotlin.text / mapNotNull
Platform and version requirements: JVM (1.0), JS (1.0), Native (1.0) 
inline fun <R : Any> CharSequence.mapNotNull(
    transform: (Char) -> R?
): List<R>

Returns a list containing only the non-null results of applying the given transform function to each character in the original char sequence.

import kotlin.test.*

fun main(args: Array<String>) {
//sampleStart
val strings: List<String> = listOf("12a", "45", "", "3")
val ints: List<Int> = strings.mapNotNull { it.toIntOrNull() }

println(ints) // [45, 3]
println(ints.sum()) // 48
//sampleEnd
}

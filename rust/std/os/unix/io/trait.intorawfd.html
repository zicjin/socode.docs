Trait std::os::unix::io::IntoRawFd

pub trait IntoRawFd {
    fn into_raw_fd(self) -> RawFd;
}
This is supported on Unix only.

A trait to express the ability to consume an object and acquire ownership of its raw file descriptor.

Required methods

[src]

fn into_raw_fd(self) -> RawFd

Consumes this object, returning the raw underlying file descriptor.

This function transfers ownership of the underlying file descriptor to the caller. Callers are then the unique owners of the file descriptor and must close the descriptor once itâ€™s no longer needed.

Example
use std::fs::File;
#[cfg(unix)]
use std::os::unix::io::{IntoRawFd, RawFd};
#[cfg(target_os = "wasi")]
use std::os::wasi::io::{IntoRawFd, RawFd};

let f = File::open("foo.txt")?;
#[cfg(any(unix, target_os = "wasi"))]
let raw_fd: RawFd = f.into_raw_fd();

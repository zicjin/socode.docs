Deno.ConnectOptions

interface ConnectOptions {
hostname?: string;

port: number;

transport?: "tcp";

}

Properties

hostname?: string

A literal IP address or host name that can be resolved to an IP address. If not specified, defaults to 127.0.0.1.

port: number

The port to connect to.

transport?: "tcp"

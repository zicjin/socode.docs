Deno.NetPermissionDescriptor

interface NetPermissionDescriptor {
host?: string;

name: "net";

}

Properties

host?: string

Optional host string of the form "<hostname>[:<port>]". Examples:

 "github.com"
 "deno.land:8080"
name: "net"

wp_check_mysql_version()

Checks the version of the installed MySQL binary.

Source

File: wp-admin/includes/upgrade.php 

function wp_check_mysql_version() {
	global $wpdb;
	$result = $wpdb->check_database_version();
	if ( is_wp_error( $result ) ) {
		wp_die( $result );
	}
}

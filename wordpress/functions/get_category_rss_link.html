function get_category_rss_link($display = false, $cat_ID = 1) {
	_deprecated_function( __FUNCTION__, '2.5.0', 'get_category_feed_link()' );

	$link = get_category_feed_link($cat_ID, 'rss2');

	if ( $display )
		echo $link;
	return $link;
}

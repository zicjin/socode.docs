Retrieves all error codes.

Return

(array) List of error codes, if available.

Source

File: wp-includes/class-wp-error.php 

public function get_error_codes() {
		if ( ! $this->has_errors() ) {
			return array();
		}

		return array_keys( $this->errors );
	}

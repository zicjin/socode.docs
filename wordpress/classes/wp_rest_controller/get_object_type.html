Retrieves the object type this controller is responsible for managing.

Return

(string) Object type for the controller.

protected function get_object_type() {
		$schema = $this->get_item_schema();

		if ( ! $schema || ! isset( $schema['title'] ) ) {
			return null;
		}

		return $schema['title'];
	}

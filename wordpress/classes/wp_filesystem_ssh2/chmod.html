WP_Filesystem_SSH2::chmod( string $file, int|false $mode = false, bool $recursive = false )

Changes filesystem permissions.

Parameters

$file

(string) (Required) Path to the file.

$mode

(int|false) (Optional) The permissions as octal number, usually 0644 for files, 0755 for directories.

Default value: false

$recursive

(bool) (Optional) If set to true, changes file permissions recursively.

Default value: false

Return

(bool) True on success, false on failure.

public function chmod( $file, $mode = false, $recursive = false ) {
		if ( ! $this->exists( $file ) ) {
			return false;
		}

		if ( ! $mode ) {
			if ( $this->is_file( $file ) ) {
				$mode = FS_CHMOD_FILE;
			} elseif ( $this->is_dir( $file ) ) {
				$mode = FS_CHMOD_DIR;
			} else {
				return false;
			}
		}

		if ( ! $recursive || ! $this->is_dir( $file ) ) {
			return $this->run_command( sprintf( 'chmod %o %s', $mode, escapeshellarg( $file ) ), true );
		}

		return $this->run_command( sprintf( 'chmod -R %o %s', $mode, escapeshellarg( $file ) ), true );
	}

WP_REST_Menu_Items_Controller::check_has_read_only_access( WP_REST_Request $request )

Checks whether the current user has read permission for the endpoint.

Description

This allows for any user that can edit_theme_options or edit any REST API available post type.

Parameters

$request

(WP_REST_Request) (Required) Full details about the request.

Return

(bool|WP_Error) Whether the current user has permission.

Source

File: wp-includes/rest-api/endpoints/class-wp-rest-menu-items-controller.php 

protected function check_has_read_only_access( $request ) {
		if ( current_user_can( 'edit_theme_options' ) ) {
			return true;
		}

		if ( current_user_can( 'edit_posts' ) ) {
			return true;
		}

		foreach ( get_post_types( array( 'show_in_rest' => true ), 'objects' ) as $post_type ) {
			if ( current_user_can( $post_type->cap->edit_posts ) ) {
				return true;
			}
		}

		return new WP_Error(
			'rest_cannot_view',
			__( 'Sorry, you are not allowed to view menu items.' ),
			array( 'status' => rest_authorization_required_code() )
		);
	}

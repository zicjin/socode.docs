fun Char.isLetterOrDigit(): Boolean

Returns true if this character is a letter or digit.

import java.util.*
import kotlin.test.*

fun main(args: Array<String>) {
//sampleStart
val chars = listOf('a', '1', '+')

println(notLetterOrDigitList) // [+]

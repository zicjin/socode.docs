Future<ScrollState> setDistributeScroll(String nativeScrollBehavior) {
  var completer = new Completer<ScrollState>();
  _setDistributeScroll((value) {
    completer.complete(value);
  }, nativeScrollBehavior);
  return completer.future;
}

void setInnerHtml(String? html,
    {NodeValidator? validator, NodeTreeSanitizer? treeSanitizer}) {
  this.nodes.clear();
  append(document.body!.createFragment(html,
      validator: validator, treeSanitizer: treeSanitizer));
}

Future<String> decodeStream(Stream<List<int>> byteStream) {
  return decoder
      .bind(byteStream)
      .fold(StringBuffer(),
          (StringBuffer buffer, String string) => buffer..write(string))
      .then((StringBuffer buffer) => buffer.toString());
}

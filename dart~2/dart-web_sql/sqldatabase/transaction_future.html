@JSName('transaction')
Future<SqlTransaction> transaction_future() {
  var completer = new Completer<SqlTransaction>();
  transaction((value) {
    applyExtension('SQLTransaction', value);
    completer.complete(value);
  }, (error) {
    completer.completeError(error);
  });
  return completer.future;
}

Examples

- name: Boot image test
  hosts: ism
  connection: local
  gather_facts: no
  vars:
    ism:
      host: "{{ ansible_ssh_host }}"
      username: "{{ username }}"
      password: "{{ password }}"

  tasks:

  - name: "Set bmc boot image"
    inspur.sm.edit_boot_image:
      image: 2
      provider: "{{ ism }}"

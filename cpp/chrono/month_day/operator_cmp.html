                          const std::chrono::month_day& y) noexcept;
 (5) (since C++20) 
constexpr bool operator>=(const std::chrono::month_day& x,
                          const std::chrono::month_day& y) noexcept;
 (6) (since C++20)

Compares two month_day objects.

Return value

1) x.month() == y.month() && x.day() == y.day()
2) !(x == y).
3) x.month() < y.month() || (x.month() == y.month() && x.day() < y.day())
4) y < x
5) !(y < x)
6) !(x < y)
